النتيجة:
اللون يدل على أن الإجابة خاطئة
اللون يدل على أن إختيار الإجابة صحيح
اللون يدل على الإجابة الصحيحة سواء تم اختيارها أولا
1_ لـي مــــن الشّمس خلعة صــفـراء
2_ ومـــن الـزمـهـريــر إن حدّث الغيـ
3_ لو تراني في الشّمس والبرد قد أنـ
4_ لــــي مــن الليل والنهار على الطو
5_ شــنـــع الــنّــاسُ أنّــنـــي جــاهـلي
6_ آه واحـسـرتـي لـقـد ذهــب الـعـمـ
7_ كـلـمـا قـلـت فـي غـد أدرك الــسـؤ
8_ يـا فـؤادي صـبـرا فما زالت الأيام
لا أبـــالـي إذا أتــــــانــــي الـشّـتاء
ـم ثـــيـــابــي وطـيـلـسانـي الهواء
ـحـل جـسـمـي لـقـلـت: إني هباء
ل عـــزاء (لا يـنـقــضــي)وهــنـاء
مــــانــــوى ومــــا لــهُـــــمْ أهْواءُ
ــر وحــــظّـــي تــأســــف وعـنـاء
ل أتــــانــي غــــــد بـمـا لا أشـــاء
فــيــهــا الــســرّاء والـــــضــــرّاء
